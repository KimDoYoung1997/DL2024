3. torch.eigensum

A. transpose

tsr = torch.tensor([1,2,3,4,5,6]).reshape(3,2)
tsr 
tensor([[1, 2],
        [3, 4],
        [5, 6]])
tsr.t()
tensor([[1, 3, 5],
        [2, 4, 6]])
torch.einsum('ij -> ji',tsr) 
tensor([[1, 3, 5],
        [2, 4, 6]])

B. 행렬곱

tsr1 = torch.tensor([1,2,3,4,5,6]).reshape(3,2)/6
tsr1 
tensor([[0.1667, 0.3333],
        [0.5000, 0.6667],
        [0.8333, 1.0000]])
tsr2 = torch.tensor([1,2.0]).reshape(2,1)
tsr2
tensor([[1.],
        [2.]])
tsr1 @ tsr2
tensor([[0.8333],
        [1.8333],
        [2.8333]])
torch.einsum('ij,jk -> ik',tsr1,tsr2) 
tensor([[0.8333],
        [1.8333],
        [2.8333]])
